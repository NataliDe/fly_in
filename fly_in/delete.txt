3. Додай функцію для формування capacity info

Встав над run_text_mode():

def format_capacity_info(simulator: Simulator) -> list[str]:
    hub_occupancy = simulator._hub_occupancy()
    link_load = simulator._active_link_load()

    lines: list[str] = []

    for hub_name in sorted(simulator.map_data.hubs):
        hub = simulator.map_data.hubs[hub_name]
        used = hub_occupancy[hub_name]

        if hub.kind in {"start", "end"}:
            capacity = "∞"
        else:
            capacity = str(hub.max_drones)

        lines.append(f"Zone {hub_name}: {used}/{capacity} drones")

    for key in sorted(simulator.map_data.connections):
        connection = simulator.map_data.connections[key]
        used = link_load[key]
        capacity = connection.max_link_capacity
        lines.append(
            f"Connection {connection.display_name()}: {used}/{capacity} capacity used"
        )

    return lines
4. Розшир print_turn_output()

Було:

def print_turn_output(simulator: Simulator, show_log: bool) -> None:
    if show_log and simulator.move_logs:
        print(" ".join(simulator.move_logs))

Стане:

def print_turn_output(
    simulator: Simulator,
    show_log: bool,
    show_capacity_info: bool,
) -> None:
    if show_log and simulator.move_logs:
        print(" ".join(simulator.move_logs))

    if show_capacity_info:
        for line in format_capacity_info(simulator):
            print(line)
